
RECRUITMENT_bender_only = {

	allowed = {
		always = yes
	}
	enable = {
		stockpile_ratio = {
			archetype = earthbender_equipment
			ratio = 0.01
		}
		OR = {
			has_idea = fire_benders_only
			has_idea = earth_benders_only
			has_idea = water_benders_only
			has_idea = air_benders_only
		}
		OR = {
			has_at_least_31_of_own_bender_equipment = yes
			check_variable = {
				var = ROOT.amount_manpower_in_deployment_queue
				value = 110
				compare = less_than_or_equals
			}
		}	
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = role_ratio
		id = benders_only
		value = 1000
	}

	ai_strategy = {
		type = role_ratio
		id = combined_arms
		value = -100
	}

	ai_strategy = {
		type = role_ratio
		id = defender
		value = -100
	}

	ai_strategy = {
		type = role_ratio
		id = garrison
		value = -100
	}
}
